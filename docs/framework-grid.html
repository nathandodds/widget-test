FRAMEWORK = GRIDS

This is the base framework for our grid layout. Please try not touch this file and change styles as this will affect how the framework will behave.

If you need to change the way a sector is styled then add a class and style that class.

@if ($grid) == true {
	.grid {
		letter-spacing: -0.28em; // makes our grid items have a true width

		& > * {
			letter-spacing: normal; // makes all the letter spacing with our grid class have letter spacing of normal
			padding: $grid-margin;
		}
	}

	// Every class that has grid__ at the start we want to apply these styles to them
	[class*='grid__'] {
		display: block;
		list-style: none;
		padding: ($grid-margin / 2) $grid-margin;
		text-align: center;
		vertical-align: top;
		width: 100%; // default all our classes that have grid__ will be 100% for our mobile

		@include bp(desktop) {
			display: inline-block;
		}
	}

	// Grid half
	.grid__half {
		@include bp(desktop) {
			width: 50%;
		}
	}

	// THIRDS
	// Grid third
	.grid__third {
		@include bp(desktop) {
			width: 33.33333%;
		}
	}

	// Grid two third
	.grid__two-third {
		@include bp(desktop) {
			width: 66.66666%;
		}
	}

	// QUARTERS
	// Grid quarter
	.grid__quarter {
		@include bp(desktop) {
			width: 25%;
		}
	}

	// Grid three quarter
	.grid__three-quarter {
		@include bp(desktop) {
			width: 75%;
		}
	}

	// FIFTHS
	// Grid fifths
	.grid__fifth {
		@include bp(desktop) {
			width: 20%;
		}
	}

	// Grid two fifths
	.grid__two-fifth {
		@include bp(desktop) {
			width: 40%;
		}
	}

	// Grid three fifths
	.grid__three-fifth {
		@include bp(desktop) {
			width: 60%;
		}
	}

	// Grid four fifths
	.grid__four-fifth {
		@include bp(desktop) {
			width: 80%;
		}
	}
}
